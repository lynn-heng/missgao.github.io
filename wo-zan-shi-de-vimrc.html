我暂时的VIMRC

用了好长时间的LINUX，话说LINUX大神门大多用VIM编辑器，我一直在尝试，但总还不是十分习惯。偶尔用之也得设置下，下面是我现在的设置，慢慢补之吧，等我的vimrc够强悍了估计VIM水平才到家。

" how many lines of history VIM has to remember
set history=700

" Enable filetype plugins
filetype plugin on
filetype indent on

" No annoying sound on errors
set noerrorbells
set novisualbell

" Set utf8 as standard encoding
set encoding=utf8

" Turn backup off, since most stuff is in SVN, git et.c anyway...
set nobackup
set nowb
set noswapfile

set number
set ruler
set tabstop=4
set linebreak
syn on
set showmatch
set autoindent
set hlsearch

#auto recognition of code
set fileencodings=ucs-bom,utf-8,cp936,gb18030,big5,euc-jp,euc-kr,latin1
